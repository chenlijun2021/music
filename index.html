        async function setupAudio() {

        // --- 核心物理心流与画作生成引擎 ---

        function animate() {

            if (!isActive) return;


            analyser.getByteFrequencyData(dataArray);

            

            let sum = 0;

            let bassSum = 0, bassCount = 0;

            let trebleSum = 0, trebleCount = 0;


            for(let i=0; i<dataArray.length; i++) {

                let val = dataArray[i];

                sum += val;

                if (i < 12) { bassSum += val; bassCount++; }           

                else if (i > 40) { trebleSum += val; trebleCount++; }  

            }

            

            let average = sum / dataArray.length;

            let bassAvg = bassCount > 0 ? bassSum / bassCount : 0;

            let trebleAvg = trebleCount > 0 ? trebleSum / trebleCount : 0;


            // 瞬态节奏爆发检测

            let volumeDelta = average - prevAverage;

            if (volumeDelta > 15 && average > strokeConfig.threshold) {

                beatInertia = Math.min(beatInertia + (volumeDelta / 10), 8.0);

            } else {

                beatInertia = Math.max(0, beatInertia - 0.15); 

            }

            prevAverage = average;


            smoothedBass += (bassAvg - smoothedBass) * 0.1;

            smoothedTreble += (trebleAvg - smoothedTreble) * 0.1;


            if (average > strokeConfig.threshold) {

                let chargeRate = (average - strokeConfig.threshold) / 20;

                flowEnergy = Math.min(1.0, flowEnergy + Math.max(0.1, chargeRate));

            } else {

                flowEnergy = Math.max(0, flowEnergy - 0.015); 

            }


            if (flowEnergy > 0.1) {

                globalTime += 0.05;

                stateTimer--;


                if (stateTimer <= 0) {

                    artistState = Math.random() < 0.5 ? 'scribble' : 'wander';

                    stateTimer = 40 + Math.floor(Math.random() * 80); 

                }


                let bassFactor = smoothedBass / 255;

                let trebleFactor = smoothedTreble / 255;


                // 1. 独立计算画笔移动速度 (受瞬时节拍推动)

                let currentSpeed = paintSpeed * (1 + beatInertia * 0.3);


                // 2. 空间曲率解绑算法 (Decoupled Spatial Curvature)

                // 先计算出“单位距离”内的角变化率 (turnRatePerPixel)，使线条几何弧度完全独立于绘画移动速度！

                let turnRatePerPixel = 0;


                // 低频振幅控制大范围转向趋势

                let macroTurn = Math.sin(globalTime * 0.15) * (0.02 + bassFactor * 0.08);


                if (artistState === 'scribble') {

                    let styleCycle = Math.sin(globalTime * 0.25); 

                    if (styleCycle > 0.4) {

                        // 局部无序涂鸦 (高频控制)

                        turnRatePerPixel = (0.35 + trebleFactor * 0.6) * Math.sign(Math.sin(globalTime * 1.1));

                    } else if (styleCycle < -0.4) {

                        // 经典大回环：即便速度慢，转角率也保持极低，拉出极大极流畅的圆圈

                        turnRatePerPixel = (0.015 + bassFactor * 0.04) * Math.sign(Math.sin(globalTime * 0.6));

                    } else {

                        // 随机手绘纠缠

                        turnRatePerPixel = (trebleFactor * 0.35) + Math.sin(globalTime * 3) * 0.12;

                    }

                } else {

                    // 藤蔓式自然漫步：大而松弛的舒展弧线

                    turnRatePerPixel = macroTurn + Math.sin(globalTime * 0.3) * (0.03 + bassFactor * 0.05); 

                }


                // 【核心解绑公式】：每帧旋转角变化 = 空间弯曲率 * 本帧行进位移！

                // 这确保了不管 currentSpeed 多慢，大圆弧的直径和形状都完全不缩水

                angle += turnRatePerPixel * (currentSpeed * 0.85);


                // 基础横向推力，温和引导大方向向右

                let baseForwardForce = paintSpeed * 0.15; 

                let targetVx = Math.cos(angle) * currentSpeed + baseForwardForce;

                let targetVy = Math.sin(angle) * currentSpeed;


                // 注入细腻的高频手绘细微颤抖

                let jitterAmount = 0.04 + trebleFactor * 1.0;

                let handJitterX = Math.sin(globalTime * 17) * jitterAmount;

                let handJitterY = Math.cos(globalTime * 13) * jitterAmount;


                // 物理缓冲平滑过渡

                let smoothing = Math.max(0.05, 0.15 - (beatInertia * 0.02)); 

                vx += (targetVx - vx) * smoothing + handJitterX;

                vy += (targetVy - vy) * smoothing + handJitterY;


                penX += vx;

                penY += vy;


                // 边界限制柔和约束

                const dpr = window.devicePixelRatio || 1;

                const logicalHeight = canvas.height / dpr;

                const margin = 180;

                if (penY < margin) {

                    vy += 0.4; angle += 0.03;

                } else if (penY > logicalHeight - margin) {

                    vy -= 0.4; angle -= 0.03;

                }

                

                if (penX < cameraX + 50) {

                    vx += 1.0; angle += (0 - angle) * 0.02;

                }


                // 笔压力 (线宽随重低音与爆发饱满度起伏)

                let dynamicWeight = strokeConfig.weight * (1 + bassFactor * 1.5 + beatInertia * 0.5);

                points.push({ x: penX, y: penY, w: dynamicWeight });


                // 相机跟随焦点

                const logicalWidth = canvas.width / dpr;

                const viewFocus = logicalWidth * 0.45;

                if (penX - cameraX > viewFocus) {

                    targetCameraX = penX - viewFocus;

                }


            } else {

                if (artistState !== 'idle') {

                    points.push(null); 

                    artistState = 'idle';

                }

                vx *= 0.9;

                vy *= 0.9;

            }


            // 平滑相机滚动：基于完全独立的【推进速度 (scrollSpeed)】变量

            let cameraEasing = Math.min(0.2, scrollSpeed * 0.03); 

            cameraX += (targetCameraX - cameraX) * cameraEasing;


            updateStateUI();

            draw();

            requestAnimationFrame(animate);

        }


        // --- 动态写意笔压渲染器 ---

        function draw() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);


            if (points.length < 2) return;


            const dpr = window.devicePixelRatio || 1;


            ctx.save();

            ctx.scale(dpr, dpr);

            ctx.translate(-cameraX, 0);


            ctx.lineCap = 'round';

            ctx.lineJoin = 'round';

            ctx.strokeStyle = 'rgba(15, 15, 15, 0.9)'; 


            let p0 = points[0];

            let p1 = points[1];

            

            if (p0 && p1) {

                let lastMidX = p0.x;

                let lastMidY = p0.y;


                for (let i = 1; i < points.length - 1; i++) {

                    let p2 = points[i + 1];


                    if (!p1 || !p2) {

                        lastMidX = null;

                        p1 = p2;

                        continue;

                    }

                    if (lastMidX === null) {

                        lastMidX = p1.x;

                        lastMidY = p1.y;

                    }


                    let midX = (p1.x + p2.x) / 2;

                    let midY = (p1.y + p2.y) / 2;


                    ctx.beginPath();

                    ctx.moveTo(lastMidX, lastMidY);

                    ctx.quadraticCurveTo(p1.x, p1.y, midX, midY);

                    

                    ctx.lineWidth = p1.w || strokeConfig.weight; 

                    ctx.stroke();


                    lastMidX = midX;

                    lastMidY = midY;

                    p1 = p2;

                }

            }


            // 标志性的朱砂红笔尖触点

            if (artistState !== 'idle' && points[points.length - 1]) {

                ctx.beginPath();

                let tipRadius = (points[points.length - 1].w * 0.6) + 1.5;

                ctx.arc(penX, penY, tipRadius, 0, Math.PI * 2);

                ctx.fillStyle = '#ef4444'; 

                ctx.fill();

            }


            ctx.restore();

        }
